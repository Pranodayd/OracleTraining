<bookstore>
	
	<book>
  		<title>Harry Potter</title>
  		<author>J K. Rowling</author>
  		<year>2005</year>
  		<price>29.99</price>
	</book>

</bookstore>

bookstore             		Will return all bookstore elements
/bookstore			Will return bookstore which is Root
bookstore/book			Will return ALL book elements which are CHILD of bookstore
bookstore//book			Will return ALL book elements which are DESCENDANT of bookstore
//@lang				Will return me element having Lang attribute set

bookstore/book[1]		Will return 1st book child element of bookstore
bookstore/book[last()]		Will return LAST book child element of bookstore.And this may return different elements
				depending on how many we have,hence it is called DYNAMIC XPATH
bookstore/book[@lang='eng']	Will return Child element of bookstore with name book having lang attribute set to 'Eng'
